	Email           AS UPPER(LEFT(first_name,1))+LOWER(LEFT(last_name,3)) +'@mail.com',

SELECT T1.FIRST_NAME
       ,CASE WHEN T2.EMPLOYEE_ID IS NULL THEN 'Currently Working'
	ELSE CONCAT('Left the company at',' ',T2.END_DATE) END AS STATUS
FROM EMPLOYEES T1
JOIN JOB_HISTORY T2 ON T1.EMPLOYEE_ID = T2.EMPLOYEE_ID
