CREATE PROCEDURE Newnumber
    @BusinessEntityID INT,
    @NationalIDNumber  NVARCHAR(20)
AS
BEGIN
    UPDATE HumanResources.Employee
    SET NationalIDNumber = @NationalIDNumber
    WHERE BusinessEntityID = @BusinessEntityID;
END;
